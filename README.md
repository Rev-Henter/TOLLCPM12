# INSTALLATION COMMANDS

━━━━━━━━━━━━━━━━━━━━  
ANDROID (TERMUX)  
━━━━━━━━━━━━━━━━━━━━  

~~~bash
termux-setup-storage
apt update
apt upgrade -y
pkg install python -y
pkg install git -y
pkg install python-pip -y
git clone https://github.com/Rev-Henter/TOLLCPM12.git
cd TOLLCPM12
pip install -r requirements.txt
python main.py
~~~

iOS (a-Shell / iSH)  
━━━━━━━━━━━━━━━━━━━━  

~~~bash
apt update
apt upgrade -y
apt install git python3 python3-pip -y
git clone https://github.com/Rev-Henter/TOLLCPM12.git
cd TOLLCPM12
pip3 install -r requirements.txt
python3 main.py
~~~
